from fastapi import FastAPI from fastapi.responses import HTMLResponse from datetime import datetime import json import html from validator import validate_signal from analytics_engine import analyze_signal from samachar_input_adapter import load_data, convert_to_signals from multi_signal_analyzer import analyze_patterns app = FastAPI() # ----------------------------- # SAFE ZONE DETECTOR # ----------------------------- def detect_zone(signal): try: lat = float(signal.get("latitude", 0)) return "Zone_A" if lat > 20 else "Zone_B" except: return "Unknown" # ----------------------------- # SAFE STRING # ----------------------------- def safe(v): if isinstance(v, dict) or isinstance(v, list): return json.dumps(v, default=str) return str(v) # ----------------------------- # DASHBOARD # ----------------------------- @app.get("/", response_class=HTMLResponse) def dashboard(): # 🔥 reload every request (important fix) weather, aqi = load_data() signals = convert_to_signals(weather, aqi) rows = "" processed_outputs = [] for signal in signals[:20]: try: validation = validate_signal(signal) if validation["status"] == "REJECT": continue analysis = analyze_signal(signal) zone = detect_zone(signal) output = { "signal_id": safe(signal.get("signal_id")), "trace_id": safe(validation.get("trace_id")), "risk_level": safe(analysis.get("risk_level")), "anomaly_type": safe(analysis.get("anomaly_type")), "zone": safe(zone) } processed_outputs.append(output) rows += f""" {safe(signal.get("signal_id"))} {safe(zone)} {safe(validation.get("status"))} {safe(analysis.get("risk_level"))} {safe(analysis.get("anomaly_type"))} {html.escape(safe(analysis.get("explanation")))} """ except Exception: continue # ----------------------------- # 🔥 SAFE PATTERN ANALYSIS INPUT # ----------------------------- safe_inputs = [ { k: safe(v) for k, v in o.items() } for o in processed_outputs ] pattern = analyze_patterns(safe_inputs) # ----------------------------- # STATS # ----------------------------- total_signals = len(signals) total_anomalies = len([o for o in processed_outputs if o.get("risk_level") != "LOW"]) html_content = f""" NICAI Dashboard

NICAI Dashboard

Pattern

ID: {pattern.get("pattern_id")}

Type: {pattern.get("pattern_type")}

Severity: {pattern.get("severity_trend")}

Stats

Total: {total_signals}

Anomalies: {total_anomalies}

Signals

{rows}
ID Zone Status Risk Anomaly Explanation Action
""" return HTMLResponse(content=html_content) # ----------------------------- # ACTION API # ----------------------------- @app.post("/action") async def action_router(payload: dict): action_payload = { "trace_id": payload.get("trace_id"), "action_type": payload.get("action_type"), "target_role": "authority", "timestamp": str(datetime.utcnow()), "context": payload.get("context", {}) } with open("action_logs.json", "a") as f: f.write(json.dumps(action_payload) + "\n") return {"status": "action logged", "payload": action_payload}
